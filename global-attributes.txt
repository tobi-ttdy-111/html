class = "nombreClase" => permiten que CSS y JavaScript seleccionen y accedan a elementos específicos
frecuencia de uso: *****
recomendada: true


dir = ( "ltr" / "rtl" / "auto" ) => indica la direccionalidad del texto del elemento
frecuencia de uso: *
recomendada: true


id = "nombreId" => define un identificador único, mismo proposito que 'class'
frecuencia de uso: *****
recomendada: true


style = estilo(s) css ( "color: red; font..." ) => declaraciones de estilo CSS que se aplicarán al elemento
frecuencia de uso: **
recomendada: false
